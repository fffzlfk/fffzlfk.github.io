### 马的遍历
<h4><a href="https://www.luogu.com.cn/problem/P1443">题目链接</a></h4>

#### AC代码
```cpp
#include <algorithm>  
#include <cmath>  
#include <cstdio>  
#include <cstring>  
#include <iostream>  
#include <queue>  
using namespace std;

struct node {
    int x;
    int y;
    node(int a, int b) {
        x = a;
        y = b;
    }
};

int dx[] = {-1, 1, 2, -2};
int dy[] = {2, -2, -1, 1};

int n, m;
bool visit[405][405];
int moves[405][405];

void bfs(int x, int y, int step) {
    queue<node> Q;
    Q.push(node(x, y));
    visit[x][y] = 1;
    moves[x][y] = step;
    while (!Q.empty()) {
        node t = Q.front();
        Q.pop();
        for (int i = 0; i < 4; i++) {
            int nx = t.x + dx[i];
            if (nx < 1 || nx > n) continue;
            for (int j = 0; j < 4; j++) {
                if (abs(dx[i]) == abs(dy[j])) continue;
                int ny = t.y + dy[j];
                if (ny >= 1 && ny <= m && !visit[nx][ny]) {
                    Q.push(node(nx, ny));
                    visit[nx][ny] = 1;
                    moves[nx][ny] = moves[t.x][t.y] + 1;
                }
            }
        }
    }
}

int main() {
    int x, y;
    scanf("%d%d%d%d", &n, &m, &x, &y);
    memset(moves, -1, sizeof(moves));
    bfs(x, y, 0);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) { printf("%-5d", moves[i][j]); }
        printf("\n");
    }
    return 0;
}
```
### 填涂颜色
<h4><a href="https://www.luogu.com.cn/problem/P1162">题目链接</a></h4>

#### AC代码
```cpp
#include <algorithm>  
#include <cmath>  
#include <cstdio>  
#include <cstring>  
#include <iostream>  
#include <queue>  
using namespace std;

int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};

int n;
int M[35][35];
void dfs(int x, int y) {
    M[x][y] = 3;
    for (int i = 0; i < 4; i++) {
        int nx = x + dx[i];
        if (nx > n + 1 || nx < 0) continue;
        int ny = y + dy[i];
        if (ny > n + 1 || ny < 0) continue;
        if (!M[nx][ny]) dfs(nx, ny);
    }
    return;
}
int main() {
    scanf("%d", &n);
    memset(M, 0, sizeof(M));
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) scanf("%d", &M[i][j]);
    }
    dfs(0, 0);
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            if (M[i][j] == 3)
                printf("0 ");
            else if (M[i][j] == 0)
                printf("2 ");
            else
                printf("%d ", M[i][j]);
        }
        printf("\n");
    }
    return 0;
}
```
### 迷宫
<h4><a href="https://www.luogu.com.cn/problem/P1605">题目链接</a></h4>

#### AC代码
```cpp
#include <cmath>  
#include <cstdio>  
#include <cstring>  
#include <iostream>  
#include <queue>  
using namespace std;

int dx[] = {1, -1, 0, 0};
int dy[] = {0, 0, 1, -1};

int n, m, t;
int x11, y11;
int x22, y22;
int M[7][7];
bool visit[7][7];
int cnt = 0;

void dfs(int x, int y) {
    visit[x][y] = 1;
    if (x == x22 && y == y22) cnt++;
    for (int i = 0; i < 4; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];
        if (nx < 1 || nx > n || ny < 1 || ny > m) continue;
        if (!M[nx][ny] && !visit[nx][ny]) { dfs(nx, ny); }
    }
    visit[x][y] = 0;
    return;
}

int main() {
    scanf("%d%d%d", &n, &m, &t);
    scanf("%d%d%d%d", &x11, &y11, &x22, &y22);
    int xx, yy;
    while (t--) {
        scanf("%d%d", &xx, &yy);
        M[xx][yy]++;
    }
    dfs(x11, y11);
    printf("%d\n", cnt);
}
```
### 数的划分
<h4><a href="https://www.luogu.com.cn/problem/P1025">题目链接</a></h4>

#### AC代码
```cpp
#include <cmath>  
#include <cstdio>  
#include <cstring>  
#include <iostream>  
#include <queue>  
#include <set>  
#define in(n) scanf("%d", &n)
using namespace std;

int func(int n, int k) {
    if (n < k) return 0;
    if (n == 1 || k == 1 || n == k) return 1;
    return func(n - 1, k - 1) + func(n - k, k);
}

int main() {
    int n, k;
    in(n);
    in(k);
    cout << func(n, k) << '\n';
    system("pause");
    return 0;
}
```

### 奇怪的电梯
<h4><a href="https://www.luogu.com.cn/problem/P1135">题目链接</a></h4>

#### AC代码
```cpp
#include <cstdio>  
#include <cstring>  
#include <iostream>  
#include <queue>  
#include <set>  
#define in(n) scanf("%d", &n)
using namespace std;

int n;
int M[205];
int a, b;
int visit[205];
int step[205];
int ans = 0x3f3f3f3f;
void bfs(int x) {
    if (x == b) {
        ans = step[x];
    }
    queue<int> Q;
    Q.push(x);
    visit[x] = 1;
    while (!Q.empty()) {
        int t = Q.front();
        Q.pop();
        if (t == b) {
            ans = step[t];
        }
        for (int i = 0; i < 2; i++) {
            int d = t + (i == 0 ? M[t] : -M[t]);
            if (d < 1 || d > n) continue;
            if (!visit[d]) {
                Q.push(d);
                visit[d] = 1;
                step[d] = step[t] + 1;
            }
        }
    }
}
int main() {
    scanf("%d%d%d", &n, &a, &b);
    for (int i = 1; i <= n; i++) scanf("%d", &M[i]);
    bfs(a);
    if (ans == 0x3f3f3f3f) ans = -1;
    cout << ans << '\n';
    return 0;
}
```
### POJ3984 迷宫问题
<h4><a href="http://poj.org/problem?id=3984">题目链接</a></h4>

#### AC代码
```cpp
#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <queue>
#include <stack>
using namespace std;

struct node {
    int x;
    int y;
    node(int a, int b) {
        x = a;
        y = b;
    }
};

struct path {
    int x, y;
};

int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

int n, m;
bool visit[7][7];
int M[7][7];
path pre[10][10];

void bfs(int x, int y) {
    queue<node> Q;
    Q.push(node(x, y));
    visit[x][y] = 1;
    while (!Q.empty()) {
        node t = Q.front();
        Q.pop();
        for (int i = 0; i < 4; i++) {
            int nx = t.x + dx[i];
            int ny = t.y + dy[i];
            if (nx < 1 || nx > 5 || M[nx][ny] || ny < 1 || ny > 5 ||
                visit[nx][ny]) {
                continue;
            }
            Q.push(node(nx, ny));
            visit[nx][ny] = 1;
            path p;
            p.x = t.x;
            p.y = t.y;
            pre[nx][ny] = p;
            if (nx == 5 && ny == 5) return;
        }
    }
}

int main() {
    int x, y;
    for (int i = 1; i <= 5; i++) {
        for (int j = 1; j <= 5; j++) scanf("%d", &M[i][j]);
    }
    bfs(1, 1);
    stack<node> S;
    int n_x = 5, n_y = 5;
    while (1) {
        S.push(node(n_x - 1, n_y - 1));
        if (n_x == 1 && n_y == 1) break;
        int _x = pre[n_x][n_y].x;
        int _y = pre[n_x][n_y].y;
        n_x = _x;
        n_y = _y;
    }
    while (!S.empty()) {
        printf("(%d, %d)\n", S.top().x, S.top().y);
        S.pop();
    }
    // system("pause");
    return 0;
}
```

### POJ2676 Sudoku
<h4><a href="http://poj.org/problem?id=2676">题目链接</a></h4>

#### AC代码
```cpp
#include <algorithm>
#include <cstdio>
#include <iostream>
#include <vector>
using namespace std;

int arc[10][10];

bool judge(int r, int c, int n) {
    for (int i = 0; i < 9; i++)
        if (arc[r][i] == n || arc[i][c] == n) return false;
    int _r = (r / 3) * 3;
    int _c = (c / 3) * 3;
    for (int i = _r; i < _r + 3; i++) {
        for (int j = _c; j < _c + 3; j++) {
            if (arc[i][j] == n) return false;
        }
    }
    return true;
}

vector<pair<int, int> > zero;

bool dfs(int idx) {
    if (idx >= (int)zero.size()) return true;
    int r = zero[idx].first;
    int c = zero[idx].second;
    for (int i = 1; i <= 9; i++) {
        if (judge(r, c, i)) {
            arc[r][c] = i;
            if (dfs(idx + 1)) return true;
            arc[r][c] = 0;  // 回溯
        }
    }
    return false;
}

int main() {
    int T;
    cin >> T;
    getchar();
    while (T--) {
        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {
                arc[i][j] = getchar() - '0';
                if (arc[i][j] == 0) zero.push_back(make_pair(i, j));
            }
            getchar();
        }
        dfs(0);
        for (int i = 0; i < 9; i++) {
            for (int j = 0; j < 9; j++) {
                printf("%d", arc[i][j]);
            }
            printf("\n");
        }
        zero.clear();
    }
    system("pause");
    return 0;
}
```
