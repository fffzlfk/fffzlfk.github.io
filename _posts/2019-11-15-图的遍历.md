### 问题描述
使用邻接表存储图，并设计深度优先搜索遍历算法和广度优先搜索遍历算法

### 代码实现
```cpp
#include <stdio.h>
#include <stdlib.h>

#define MAX_VERTEX_NUM 20
typedef int InfoType;
typedef int VertexType;
bool visited[MAX_VERTEX_NUM];
bool visited1[MAX_VERTEX_NUM];
typedef struct ArcNode {
	int adjvex;
	ArcNode *nextarc;
	InfoType *info;
} ArcNode;

typedef struct VNode {
	VertexType data;
	ArcNode *firstarc;
} VNode, AdjList[MAX_VERTEX_NUM];

typedef struct {
	AdjList vertices;
	int vexnum, arcnum;
	int kind;
} ALGraph;
//创建无向图
void Create(ALGraph &G) {
	puts("输入节点数和边数");
	scanf("%d%d", &G.vexnum, &G.arcnum);
	puts("依次输入结点编号"); //用1,2,3...,n表示
	for (int i = 0; i < G.vexnum; i++) {
		scanf("%d", &G.vertices[i].data);
		G.vertices[i].firstarc = NULL;
	}
	puts("依次输入弧");
	for (int i = 0; i < G.arcnum; i++) {
		int v1, v2;
		scanf("%d%d", &v1, &v2);
		v1--;
		v2--;
		ArcNode *p = (ArcNode*)malloc(sizeof(ArcNode));
		p->adjvex = v2;
		p->nextarc = G.vertices[v1].firstarc;
		G.vertices[v1].firstarc = p;
		p = (ArcNode*)malloc(sizeof(ArcNode));
		p->adjvex = v1;
		p->nextarc = G.vertices[v2].firstarc;
		G.vertices[v2].firstarc = p;
	}
}

void DFS(ALGraph *G, int v) {
	ArcNode *p;
	visited[v] = 1;
	printf("%d\n", G->vertices[v].data);
	p = G->vertices[v].firstarc;
	while (p) {
		if (!visited[p->adjvex])
			DFS(G, p->adjvex);
		p = p->nextarc;
	}
}

void DFS_traverse(ALGraph *G) {
	puts("DFS");
	int v;
	for (v = 0; v < G->vexnum; v++)
		visited[v] = 0;
	for (v = 0; v < G->vexnum; v++) {
		if (!visited[v])
			DFS(G, v);
	}
}

void BFS_traverse(ALGraph *G) {
	puts("BFS");
	int Q[MAX_VERTEX_NUM];
	int front = 0;
	int rear = 0;
	for (int k = 0; k < G->vexnum; k++)
		visited1[k] = 0;
	for (int k = 0; k < G->vexnum; k++) {
		int v = G->vertices[k].data-1;
		if (!visited1[v]) {
			Q[++rear] = v;
			while (front != rear) {
				int w = Q[++front];
				if (!visited1[w])
					printf("%d\n", G->vertices[w].data);
				visited1[w] = 1;
				ArcNode *p = G->vertices[w].firstarc;
				while (p) {
					if (!visited1[p->adjvex])
						Q[++rear] = p->adjvex;
					p = p->nextarc;
				}
			}
		}
	}
}

int main(int argc, char const *argv[]) {
	ALGraph G;
	Create(G);
	DFS_traverse(&G);
	BFS_traverse(&G);
	return 0;
}
```
